//Математический модуль Площади геометрических фигур на плоскости.

//Создан для нахождения площадей следующих геометрических фигур: треугольник, трапеция, круговой сектор
//Имеет в себе три функции treug, trap, sector

//treug - Функция, находящая площадь треугольника. Принимает 4 параметра вещественного типа: a b c - длины сторон и R - радиус описаной окружности. Возвращает площадь треугольника real.   

//trap - Функция, находящая площадь трапеции. Принимает 3 параметра вещественного типа: a b - длины оснований и h - высота. Возвращает площадь трапеции real.    

//sector - Функция, Находящая площадь сектора. Принимает 2 параметра вещественного типа: a - градусная мера сектора (0-360 градусов) и R - радиус круга. Возвращает площадь сектора real.   

unit ploshadi;

interface
///Функция, находящая площадь треугольника. Принимает 4 параметра: a b c - длины сторон и R - радиус описаной окружности. Возвращает площадь треугольника real.   
function treug(a,b,c,R: real):real;
///Функция, находящая площадь трапеции. Принимает 3 параметра: a b - длины оснований и h - высота. Возвращает площадь трапеции real.    
function trap(a,b,h: real):real;
///Функция, Находящая площадь сектора. Принимает 2 параметра a - градусная мера сектора (0-360 градусов) и R - радиус круга. Возвращает площадь сектора real.   
function sector(R,a:real):real;

implementation

function treug(a,b,c,R: real):real;
begin
  if R>0 then begin
    writeln('Площадь треугольника: ',(a*b*c)/(4*R));
    treug:=(a*b*c)/(4*R);
  end
  else begin
    writeln('Радиус описаной окружности R не быть меньше или равен нулю!');
  end;
end;

function trap(a,b,h: real):real;
begin
  writeln('Площадь трапеции: ',((a+b)/2)*h);
  trap:=((a+b)/2)*h;
end;

function sector(R,a: real):real;
begin
  if (a>=0) and (a<=360) then begin
    writeln('Площадь сектора: ',pi*R*R*(a/360));
    sector:=pi*R*R*(a/360);
  end
  else begin
    writeln('Градусная мера а должна лежать в пределах от 0 до 360!');
  end;
end;
end.
